varying vec3 vPosition;
varying vec3 vNormal;
uniform float uTime;

void main() {
  vec3 viewDirection = normalize(cameraPosition - vPosition);
  float fresnel = pow(1.0 - dot(vNormal, viewDirection), 2.0);
  
  vec3 color = mix(
    vec3(0.2, 0.8, 1.0), 
    vec3(1.0, 0.4, 0.8),
